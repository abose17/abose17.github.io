// Generate word cloud
function generateWordCloud() {
  // Count word frequencies
  const wordCounts = {};
  publicationsData.forEach(pub => {
    pub.topics.forEach(topic => {
      wordCounts[topic] = (wordCounts[topic] || 0) + 1;
    });
  });

  const words = Object.keys(wordCounts).map(word => ({
    text: word,
    size: 10 + wordCounts[word] * 5
  }));

  const color = d3.scaleOrdinal(d3.schemeCategory10);
  
  const layout = d3.layout.cloud()
    .size([500, 300])
    .words(words)
    .padding(5)
    .rotate(() => ~~(Math.random() * 2) * 90)
    .fontSize(d => d.size)
    .on('end', draw);

  function draw(words) {
    d3.select('#wordcloud')
      .append('svg')
      .attr('width', 500)
      .attr('height', 300)
      .append('g')
      .attr('transform', 'translate(250,150)')
      .selectAll('text')
      .data(words)
      .enter().append('text')
      .style('font-size', d => `${d.size}px`)
      .style('fill', (d, i) => color(i))
      .attr('text-anchor', 'middle')
      .attr('transform', d => `translate(${d.x}, ${d.y})rotate(${d.rotate})`)
      .text(d => d.text)
      .on('click', function(event, d) {
        // Filter publications by clicked topic
        const topic = d.text.toLowerCase();
        filterPublications('topic', topic);
      });
  }

  layout.start();
}

// Generate timeline
function generateTimeline() {
  // Count publications by year
  const yearCounts = {};
  const years = [];
  
  publicationsData.forEach(pub => {
    yearCounts[pub.year] = (yearCounts[pub.year] || 0) + 1;
    if (!years.includes(pub.year)) {
      years.push(pub.year);
    }
  });

  years.sort((a, b) => a - b);
  
  const margin = {top: 30, right: 30, bottom: 40, left: 50};
  const width = 500 - margin.left - margin.right;
  const height = 250 - margin.top - margin.bottom;

  const svg = d3.select('#timeline')
    .append('svg')
    .attr('width', width + margin.left + margin.right)
    .attr('height', height + margin.top + margin.bottom)
    .append('g')
    .attr('transform', `translate(${margin.left},${margin.top})`);

  const x = d3.scaleBand()
    .domain(years)
    .range([0, width])
    .padding(0.4);

  const y = d3.scaleLinear()
    .domain([0, d3.max(Object.values(yearCounts)) + 1])
    .nice()
    .range([height, 0]);

  // Add X axis
  svg.append('g')
    .attr('class', 'timeline-axis')
    .attr('transform', `translate(0,${height})`)
    .call(d3.axisBottom(x));

  // Add Y axis
  svg.append('g')
    .attr('class', 'timeline-axis')
    .call(d3.axisLeft(y).ticks(5));

  // Add bars
  svg.selectAll('.timeline-bar')
    .data(years)
    .enter()
    .append('rect')
    .attr('class', 'timeline-bar')
    .attr('x', d => x(d))
    .attr('y', d => y(yearCounts[d] || 0))
    .attr('width', x.bandwidth())
    .attr('height', d => height - y(yearCounts[d] || 0))
    .on('click', function(event, d) {
      // Filter publications by year
      filterPublications('year', d);
    });

  // Add labels
  svg.selectAll('.timeline-label')
    .data(years)
    .enter()
    .append('text')
    .attr('class', 'timeline-label')
    .attr('x', d => x(d) + x.bandwidth() / 2)
    .attr('y', d => y(yearCounts[d] || 0) - 5)
    .text(d => yearCounts[d] || 0);
}

// Filter publications based on criteria
function filterPublications(filterType, value) {
  const items = document.querySelectorAll('.publication-item');
  
  items.forEach(item => {
    if (filterType === 'all') {
      item.style.display = 'block';
    } else if (filterType === 'topic') {
      const pubId = item.dataset.id;
      const pub = publicationsData.find(p => p.id === pubId);
      if (pub && pub.topics.some(t => t.toLowerCase() === value)) {
        item.style.display = 'block';
      } else {
        item.style.display = 'none';
      }
    } else {
      if (item.dataset[filterType] === value.toString()) {
        item.style.display = 'block';
      } else {
        item.style.display = 'none';
      }
    }
  });
  
  // Update active filter button
  document.querySelectorAll('.filter-btn').forEach(btn => {
    if (filterType === 'all' && btn.dataset.filter === 'all') {
      btn.classList.add('active');
    } else if (btn.dataset.filter === filterType && btn.dataset.value === value) {
      btn.classList.add('active');
    } else {
      btn.classList.remove('active');
    }
  });
}

    }
  });
